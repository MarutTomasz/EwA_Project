// DEKLARACJA ZMIENNYCH REGULATORA PID
int Kp = 1;	 //Wzmocnienie cz³onu P
int Ti = 10; 	//Sta³a czasowa ca³kowania
int Ki = 20;	//Wzmocnienie cz³onu I
int Td = 1; 	//Sta³a czasowa ró¿niczkowania
int Kd = 1; 	//Wzmocnienie cz³onu D
int dt = 1; 	//Czas zmian wielkoœci

void ADC_init() {   // funkcja dokonuj¹ca konfiguracji przetwornika Analogowo cyfrowego
	// Ustawienie Ÿród³a referencyjnego na AVCC = 5.0 V
	sbi(ADMUX,REFS0);

	// Ustawienie podzielnika czêstotliwoœci
	sbi(ADCSRA,ADPS0);
	sbi(ADCSRA,ADPS1);
	sbi(ADCSRA,ADPS2);

	// Uruchomienie uk³adu przetwornika
	sbi(ADCSRA,ADEN);
}

uint32_t ADC_10bit() { //Funkcja dokonuj¹ca pomiaru
	// Bit rozpoczynaj¹cy konwersjê
	sbi(ADCSRA,ADSC);

	// Oczekiwanie na wyzerowanie bitu konwersji
	loop_until_bit_is_clear(ADCSRA,ADSC);

	// Zwracanie zmierzonej wartoœci
	return ADC;
}

uint32_t ADC_measure() { // Funkcja przetabiaj¹ca wartoœc ADC na Volty
	uint32_t pomiar = ADC_10bit();  // wykonanie pomiaru
	uint32_t wynik = pomiar * 500 / 1024;  // przerobienie pomiaru na wolty

	return wynik;
}

void Przerwania_Zewnetrzne() {
	sbi(GICR,INT1); // w³¹czanie zewnêtrznego przerwania na INT1
	sbi(GICR,INT0); // w³¹czanie zewnêtrznego przerwania na INT0

	sbi(MCUCR,ISC10); // zmiana stanu logicznego na INT1 rozpoczyna przerwanie
	sbi(MCUCR,ISC00); // zmiana stanu logicznego na INT0 rozpoczyna przerwanie
}

void TimerInit_PWM_1() {
	sbi(TCCR0,WGM01);  // Ustawienie trybu pracy na szybkie PWM
	sbi(TCCR0,WGM00);

	sbi(TCCR0,COM01);  // Ustawienie PWM, pojawieania siê sygna³u
	sbi(TCCR0,COM00);

	sbi(TCCR0,CS02);
//	sbi(TCCR0,CS01);
//	sbi(TCCR0,CS00); // PRESKALER 256


	OCR0 = 120;  //Wype³nienie sygna³u PWM
}

void TimerInit_PWM_2() {
	sbi(TCCR2,WGM21);// Ustawienie trybu pracy na szybkie PWM
	sbi(TCCR2,WGM20);

	sbi(TCCR2,COM21);// Ustawienie PWM, pojawieania siê sygna³u
	sbi(TCCR2,COM20);

	sbi(TCCR2,CS22);
	sbi(TCCR2,CS21);
//	sbi(TCCR2,CS20); // PRESKALER 256


	OCR2 = 120;//Wype³nienie PWM
}

//Inicjalizacja Timer1 do wywolywania przerwania z czêstotliwoœci¹ 10Hz
void TimerInit_PRZERWANIA() {
	sbi(TCCR1B,WGM12);

	sbi(TCCR1B,CS12);
//	sbi(TCCR1B,CS11);
//	sbi(TCCR1B,CS10); // PRESKALER 256


	OCR1A = 3125; // 8 000 000 / 256 (dzielnik) / 10 (10 razy na sekundê) = 3125
	sbi(TIMSK,OCIE1A);
}







// PROGRAM GLOWNY --------------------------------------------------------------------------------------------------------------------------------------------------------------------
int main() {
	DDRD = 0xff;
	PORTD = 0x00;

	cbi(DDRD,PD3);
	sbi(PORTD,PD3);

	cbi(DDRD,PD2);
	sbi(PORTD,PD2);

	DDRB = 0xff;
	PORTB = 0x00;

	Przerwania_Zewnetrzne();
	TimerInit_PWM_1();
	TimerInit_PWM_2();
	TimerInit_PRZERWANIA();

	sbi(PORTA,PA0);
	sbi(PORTA,PA1);
	ADC_init();   // inicjalizacja przetwornika

	sei();

	while(1) {
		_delay_ms(50);

	}
}

//Funkcja uruchamiana z przerwaniem po przepelnieniu licznika w timer1
ISR(TIMER1_COMPA_vect) {
	cbi(ADMUX,MUX0);
	wartosc_zadana = ADC_measure();
	sbi(ADMUX,MUX0);
	wartosc_rzeczywista = ADC_measure();

	en = wartosc_zadana/10 - wartosc_rzeczywista/10; // -16

	if (C + ((ep + en)/2)*dt < 500) //-30
		if (C + ((ep + en)/2)*dt > -500)   //-30
			C = C + ((ep + en)/2)*dt;

	U = Kp*en + (C/Ki) + Kd*(en-ep)/dt; //
	ep = en;
	if (OCR2-U > 0)
		if (OCR2-U < 255){
			OCR2 -= U;
			OCR0 -=U;
		}
}
