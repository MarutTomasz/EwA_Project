void USART_Init(){
	/*Set baud rate */
	UBRRH = (MYUBRR >> 8); // Nie mam zielonego pojêcia co to, ale instrukcja tak ka¿e
	UBRRL = MYUBRR;   // Tak te¿

	UCSRB |= (1 << RXEN) | (1 << TXEN);      // Enable receiver and transmitter
	UCSRB |= (1 << RXCIE);                   // Enable the receiver interrupt
	UCSRB |= (1 << TXCIE);                  // Enable the transmit interrupt
	UCSRC |= (1 << URSEL) |(1 << UCSZ1) | (1 << UCSZ0);    // Set frame: 8data, 1 stp

}

void USART_Transmit(unsigned char data) {
	while ( !( UCSRA & (1<<UDRE)) );// Nie wiem czy to zadzia³a, bo nie umiem wysy³ac. // Czekaj na opró¿nienie budora transmicji
	UDR = data; // W³ó¿ dane do UDR, wyœlij dane
}

unsigned char USART_Receive () {
	while ( !(UCSRA & 0b10000000));
	return UDR;	// odbierz dane z bufora
}

	TCCR1B |= (1<<CS10) | (1<<CS11); //ustawienie preskalera na 64
	TCCR1A |= (1<<COM1A0);

	USART_Init(); // inicjalizacja komunikacji

	while (1)
	{
		// Start an ADC conversion by setting ADSC bit (bit 6)
		ADCSRA = ADCSRA | (1 << ADSC);
		// Wait until the ADSC bit has been cleared
		while(ADCSRA & (1 << ADSC));

		OCR1A = 12500 - (10*ADC);

		switch (rx_byte) {
			case 'a':
				PORTB = PORTB | (1 << PB4);

		}

	}
}
